  <script>

    let currentStep = 1;


    function showStep(step) {

      document.querySelectorAll('.form-step').forEach((stepDiv) => {

        stepDiv.classList.remove('active');

      });

      document.getElementById(`step-${step}`).classList.add('active');

    }


    function nextStep(step) {

      currentStep = step;

      showStep(currentStep);

    }


    function prevStep(step) {

      currentStep = step;

      showStep(currentStep);

    }


    document.getElementById('checkoutForm').addEventListener('submit', function(e) {

      e.preventDefault();


      const requisitionDetails = document.getElementById('requisitionDetails').value;

      const bookOrderDetails = document.getElementById('bookOrderDetails').value;

      const checkoutDetails = document.getElementById('checkoutDetails').value;


      alert(`Thank you for completing the process! Here are the details:\n

        Requisition Form: ${requisitionDetails}\n

        Book Order Form: ${bookOrderDetails}\n

        Classroom Checkout Form: ${checkoutDetails}`);
